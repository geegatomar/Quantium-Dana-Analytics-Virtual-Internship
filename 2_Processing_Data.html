import numpy as np

store77 = []
store86 = []
store88 = []

for store in range(1, 273) :#and store not in {77, 86, 88}:
    if store in {77, 86, 88}:
        continue
    s = grouped.loc[(store, ), :].TOT_SALES
    s77 = grouped.loc[(77, ), :].TOT_SALES
    s86 = grouped.loc[(86, ), :].TOT_SALES
    s88 = grouped.loc[(88, ), :].TOT_SALES
    
    
    l = grouped.loc[(store, ), :].PROD_QTY
    l77 = grouped.loc[(77, ), :].PROD_QTY
    l86 = grouped.loc[(86, ), :].PROD_QTY
    l88 = grouped.loc[(88, ), :].PROD_QTY
    
    try:
        store77.append((store, pearsonr(s, s77)[0], pearsonr(l, l77)[0]))
        store86.append((store, pearsonr(s, s86)[0], pearsonr(l, l86)[0]))
        store88.append((store, pearsonr(s, s88)[0], pearsonr(l, l88)[0]))
    except:
        pass
